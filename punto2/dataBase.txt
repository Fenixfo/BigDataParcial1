CREATE EXTERNAL TABLE IF NOT EXISTS noticia.periodico (
  `category` string,
  `headline` string,
  `link` string 
) PARTITIONED BY (
  periodico string,
  year int,
  month int,
  day int 
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'
WITH SERDEPROPERTIES (
  'serialization.format' = ',',
  'field.delim' = ','
) LOCATION 's3://parcialbigdata1-2/headlines/final/'
TBLPROPERTIES ('has_encrypted_data'='false');
