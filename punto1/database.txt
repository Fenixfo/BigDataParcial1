CREATE EXTERNAL TABLE IF NOT EXISTS punto1.actions (
  `date` date,
  `open` double,
  `high` double,
  `low` double,
  `close` double,
  `adj-close` double,
  `volume` int 
) PARTITIONED BY (
  company string,
  year int,
  month int,
  day int 
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'
WITH SERDEPROPERTIES (
  'serialization.format' = ',',
  'field.delim' = ','
) LOCATION 's3://parcialbigdata1/parcialbigdata1/stocks/'
TBLPROPERTIES ('has_encrypted_data'='false');
